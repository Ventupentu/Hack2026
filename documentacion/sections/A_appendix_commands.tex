\section*{Apendice: Comandos Reproducibles}

\subsection{Instalacion}
\begin{verbatim}
pip install -r requirements.txt
pip install open_clip_torch ultralyticsplus
# opcional: pip install faiss-cpu
\end{verbatim}

\subsection{Descarga de Imagenes}
\begin{verbatim}
bash download.sh
\end{verbatim}

\subsection{Preprocesamiento}
\begin{verbatim}
python preprocess_data.py \
  --skip_download \
  --out_dir data \
  --val_ratio 0.1 \
  --seed 42
\end{verbatim}

\subsection{Generacion de Metadata de Genero}
\begin{verbatim}
python src/utils/add_gender.py
\end{verbatim}

\subsection{Entrenamiento (Hydra)}
\begin{verbatim}
python -m src.train
\end{verbatim}

\subsection{Inferencia Principal}
\begin{verbatim}
python -m src.infer infer.checkpoint_path=<ruta_a_best.pt>
\end{verbatim}

\subsection{Inferencia v8}
\begin{verbatim}
python -m src.new_infer infer.checkpoint_path=<ruta_a_best.pt>
\end{verbatim}

\subsection{Inferencia Phase 1}
\begin{verbatim}
python -m src.infer_phase1 infer.checkpoint_path=<ruta_a_best.pt>
\end{verbatim}

\subsection{Inferencia Top-1}
\begin{verbatim}
python -m src.infer_top1 infer.checkpoint_path=<ruta_a_best.pt>
\end{verbatim}

\subsection{Script Legacy OpenCLIP}
\begin{verbatim}
python -m src.infer_openclip \
  --checkpoint outputs/retrieval_openclip/best.pt \
  --submission-out outputs/retrieval_openclip/submission.csv
\end{verbatim}

\subsection{Reporte de Timestamps}
\begin{verbatim}
python src/utils/check_link_timestamps.py \
  --out-csv outputs/ts_date_alignment_report.csv \
  --timezone UTC
\end{verbatim}
